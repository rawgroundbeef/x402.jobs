```typescript
import { Router } from "express";
import type { Router as RouterType } from "express";
import { OpenFacilitator } from "@openfacilitator/sdk";
import { config } from "../config";
import { dollarsToAtomicString } from "../lib/usdc-amount";

export const x402FeesRouter: RouterType = Router();

// Reuse the same FACILITATOR_URL the rest of the API (instant.ts, webhooks.ts)
// already uses. No new env var per D-00b.
const FACILITATOR_URL = process.env.FACILITATOR_URL;
if (!FACILITATOR_URL) {
  console.warn("FACILITATOR_URL not configured — fee endpoint will not work");
}

const facilitator = FACILITATOR_URL
  ? new OpenFacilitator({ url: FACILITATOR_URL, timeout: 60000 })
  : null;

// CAIP-2 network identifiers (matches instant.ts:238-241)
const CAIP2_NETWORK_IDS = {
  solana: "solana:5eykt4UsFv8P8NJdTREpY1vzqKqZKvdp",
  base: "eip155:8453",
} as const;

// USDC addresses (matches instant.ts:30-33)
const USDC_ADDRESSES = {
  solana: "EPjFWdd5AufqSSqeM2qN1xzybapC8G4wEGGkZwyTDt1v",
  base: "0x833589fCD6eDb6E08f4c7C32D4f71b54bdA02913",
} as const;

type Network = "solana" | "base";

// Sanity cap (RESEARCH §9 risk 3): reject absurd amounts a buggy caller might submit.
const MAX_AMOUNT_USDC = 100;
const MIN_AMOUNT_USDC = 0.01;

function build402FeeResponse(
  network: Network,
  amountUsdc: number,
  payTo: string,
  resourceUrl: string,
) {
  const priceAtomic = dollarsToAtomicString(amountUsdc);
  return {
    x402Version: 2,
    error: "Payment required",
    service: { name: "x402.jobs", url: "https://x402.jobs" },
    accepts: [
      {
        scheme: "exact",
        network: CAIP2_NETWORK_IDS[network],
        amount: priceAtomic,
        maxAmountRequired: priceAtomic,
        resource: resourceUrl,
        description: `x402.jobs platform fee (${network})`,
        mimeType: "application/json",
        payTo,
        maxTimeoutSeconds: 300,
        asset: USDC_ADDRESSES[network],
        outputSchema: {
          input: { type: "http", method: "POST", bodyType: "json", bodyFields: {} },
          output: {},
        },
        extra: {
          resourceType: "platform-fee",
          serviceName: "x402.jobs",
          serviceUrl: "https://x402.jobs",
          ...(network === "solana"
            ? {
                feePayer:
                  process.env.SOLANA_FACILITATOR_ADDRESS ||
                  "561oabzy81vXYYbs1ZHR1bvpiEr6Nbfd6PGTxPshoz4p",
              }
            : {}),
        },
      },
    ],
  };
}

async function verifyAndSettleFeePayment(
  paymentHeader: string,
  amountUsdc: number,
  payTo: string,
  network: Network,
  resourceUrl: string,
): Promise<{ success: boolean; txHash?: string; error?: string }> {
  if (!facilitator) {
    return { success: false, error: "Facilitator not configured" };
  }
  try {
    const priceAtomic = dollarsToAtomicString(amountUsdc);

    let paymentPayload: any;
    try {
      paymentPayload = JSON.parse(
        Buffer.from(paymentHeader, "base64").toString("utf-8"),
      );
    } catch {
      return { success: false, error: "Invalid payment header format" };
    }

    const payment = {
      x402Version: paymentPayload.x402Version || 1,
      scheme: paymentPayload.scheme || "exact",
      network,
      payload: paymentPayload.payload,
    };

    const paymentRequirements = {
      scheme: "exact" as const,
      network,
      maxAmountRequired: priceAtomic,
      resource: resourceUrl,
      description: `x402.jobs platform fee (${network})`,
      mimeType: "application/json",
      payTo,
      maxTimeoutSeconds: 300,
      asset: USDC_ADDRESSES[network],
    };

    const result = await facilitator.settle(payment, paymentRequirements);

    if (!result.success) {
      return {
        success: false,
        error:
          (result as any).errorReason ||
          (result as any).error ||
          (result as any).message ||
          "Payment settlement failed",
      };
    }

    return { success: true, txHash: (result as any).transaction || "facilitator-settled" };
  } catch (error: any) {
    return { success: false, error: error.message };
  }
}

x402FeesRouter.post("/:network/charge", async (req, res) => {
  const network = req.params.network as Network;
  if (network !== "solana" && network !== "base") {
    return res.status(404).json({ error: `Unsupported network: ${network}` });
  }

  const payTo =
    network === "solana"
      ? process.env.FEE_COLLECTION_SOLANA_ADDRESS
      : process.env.FEE_COLLECTION_BASE_ADDRESS;
  if (!payTo) {
    return res.status(503).json({
      error: `Fee collection address not configured for ${network}`,
    });
  }

  const resourceUrl = `${config.publicUrl || process.env.PUBLIC_URL || "http://localhost:3011"}/x402/fees/${network}/charge`;

  const rawAmount = req.body && req.body.amount_usdc;
  const parsed = parseFloat(rawAmount);
  const amountUsdcRaw = Number.isFinite(parsed) ? parsed : MIN_AMOUNT_USDC;
  const amountUsdc = Math.min(Math.max(amountUsdcRaw, MIN_AMOUNT_USDC), MAX_AMOUNT_USDC);

  const paymentHeader =
    (req.headers["x-payment"] as string) ||
    (req.headers["payment-signature"] as string);

  if (!paymentHeader) {
    return res
      .status(402)
      .json(build402FeeResponse(network, amountUsdc, payTo, resourceUrl));
  }

  const result = await verifyAndSettleFeePayment(
    paymentHeader,
    amountUsdc,
    payTo,
    network,
    resourceUrl,
  );

  if (!result.success) {
    return res.status(402).json({
      x402Version: 1,
      error: result.error || "Payment failed",
      accepts: build402FeeResponse(network, amountUsdc, payTo, resourceUrl).accepts,
    });
  }

  return res.status(200).json({
    success: true,
    transactionSignature: result.txHash,
    amountPaid: amountUsdc,
    network,
  });
});
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Public Internet → `/x402/fees/:network/charge` | The endpoint is reachable without auth; x402 payment is the auth gate. |
| Caller-supplied request body → fee amount | A buggy or malicious caller could submit a huge `amount_usdc` and inflate the wallet (donation-only — they cannot drain). |
| `process.env.FEE_COLLECTION_{SOLANA,BASE}_ADDRESS` → `payTo` field in 402 response | Operator misconfiguration of these env vars routes fees to a wrong recipient. |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation Plan |
|-----------|----------|-----------|----------|-------------|-----------------|
| T-32-01 | Tampering / Repudiation | Caller-supplied `amount_usdc` could be forged below the platform's intended fee, causing under-payment | HIGH | mitigate | The handler echoes the caller-supplied amount into `accepts[0].amount`. The OpenFacilitator `settle()` call validates the on-chain transfer against the **published** `accepts[0].amount` — so the caller pays exactly what the 402 advertised. A caller who tries to pay LESS produces a settle-time error (`settle()` returns `{ success: false, error: ... }`) which the handler maps to HTTP 402 with the corrected `accepts[]`. Independent sanity cap `MAX_AMOUNT_USDC = 100` blocks absurd values at the upstream side (rejecting accidental 1000 USDC charges from a buggy `chargePlatformFee` caller). Documented in the unit test "caps amount_usdc at 100". Per RESEARCH §3.2 / §9 OQ-1 A2. |
| T-32-02 | Denial of Service | An unauthenticated caller could hammer `/x402/fees/*` with verify/settle requests | LOW | accept | x402 IS the auth (RESEARCH §9 OQ-2 / A5): each request requires real USDC to settle. Random callers can DONATE, not drain. Existing Express middleware on `apps/api/src/index.ts` (CORS, JSON body parser) covers `/x402/*` uniformly with `/`. No new rate-limiting middleware is added (matches `instant.ts` posture). Open the door to `express-rate-limit` only if/when production logs show abuse. |
| T-32-05 | Tampering | Empty / typo'd `FEE_COLLECTION_{SOLANA,BASE}_ADDRESS` routes fees to a wrong recipient | LOW | mitigate | Handler returns HTTP 503 if `payTo` is empty (Task 2, line `if (!payTo)`). Startup warning (Plan 01 Task 2) surfaces unset env vars in the API logs. Operator-side: `FEE_COLLECTION_{SOLANA,BASE}_ADDRESS` provisioning is a `STATE.md` manual task tied to wallet provenance (cold-storage / multisig, per FEE-06). Address format guidance lives in `env.example` (Wave 5 Plan 05): EIP-55 checksum for Base. |
</threat_model>
