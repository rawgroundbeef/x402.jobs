```typescript
export async function chargePlatformFee(
  params: ChargeFeeParams,
): Promise<ChargeFeeResult> {
  // Read fee config from snapshot first; fall back to live config (rows predating migration 011).
  const fee = params.runSnapshot?.config ?? {
    url: config.platformFee.resourceUrl,
    percentage: config.platformFee.percentage,
    minimum_usdc: config.platformFee.minimumUsdc,
  };

  if (!config.platformFee.enabled) {  // gate still on LIVE `enabled` (intentional: kill switch)
    return { success: true, amountPaid: 0 };
  }

  const resourceCost = params.resourceCost || 0;
  const amountUsdc = Math.max(resourceCost * fee.percentage, fee.minimum_usdc);

  // For base, derive the URL from the (possibly snapshotted) Solana URL.
  // This preserves the existing string-replace behavior for legacy snapshots
  // that stored only the Solana URL. New snapshots from Wave 1 onward record
  // whatever resourceUrl the live config had at INSERT time (still the Solana
  // form by convention).
  const resourceUrl =
    params.network === "base" && fee.url.includes("/solana/")
      ? fee.url.replace("/solana/", "/base/")
      : fee.url;

  // ... rest of existing function body: build executeX402Request payload,
  // call it, return success/error. Pass `resourceUrl` (computed above) instead
  // of the previous `config.platformFee.resourceUrl[.replace(...)]` reads.
}
```
