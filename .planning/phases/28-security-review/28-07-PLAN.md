    ```ts
    import { inngest } from '../client';
    import { supabase } from '../../lib/supabase';

    export const hardDeleteStaleUsers = inngest.createFunction(
      { id: 'hard-delete-stale-users', name: 'Hard-delete users soft-deleted >30 days ago' },
      { cron: '0 3 * * *' }, // 3 AM UTC daily
      async ({ step }) => {
        const cutoff = await step.run('compute-cutoff', () => {
          return new Date(Date.now() - 30 * 24 * 60 * 60 * 1000).toISOString();
        });

        const { data: stale, error: selErr } = await supabase
          .from('x402_users')
          .select('id, deleted_at')
          .lt('deleted_at', cutoff)
          .not('deleted_at', 'is', null);

        if (selErr) throw selErr;
        if (!stale || stale.length === 0) return { hardDeleted: 0 };

        const ids = stale.map((u) => u.id);
        const { error: delErr } = await supabase
          .from('x402_users')
          .delete()
          .in('id', ids);

        if (delErr) throw delErr;
        return { hardDeleted: ids.length };
      },
    );
    ```
