    ```ts
    /**
     * Convert USDC dollar amount (number) to atomic-unit string for facilitator BigInt parsing.
     * Closes HIGH-08: raw `expectedAmount * 1_000_000` produces decimal strings (e.g. "123456.789")
     * for fractional creator markups, which break BigInt parsing downstream.
     *
     * 1 USDC = 1,000,000 atomic units (6 decimals).
     */
    export function dollarsToAtomicString(dollars: number): string {
      if (typeof dollars !== 'number' || !Number.isFinite(dollars)) {
        throw new TypeError(`dollarsToAtomicString: expected finite number, got ${dollars}`);
      }
      if (dollars < 0) {
        throw new RangeError(`dollarsToAtomicString: negative amount not allowed: ${dollars}`);
      }
      const atomic = Math.round(dollars * 1_000_000);
      return String(atomic);
    }
    ```
