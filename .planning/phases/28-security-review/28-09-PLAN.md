<objective>
Remediate HIGH-13 by migrating the three SSRF-protected call sites from the custom `safeFetch` wrapper (introduced in CRIT-07) to a battle-tested library: `axios` + `request-filtering-agent`. The new library filters at CONNECT time (closes the DNS-rebinding/TOCTOU window left open by upfront DNS lookups), removes ~155 LOC of custom security code, and replaces it with maintained well-audited code.

Purpose: CRIT-07's `safeFetch` closed the obvious attack paths (cloud metadata, RFC1918, loopback) but leaves a DNS-rebinding window — an attacker who controls DNS can return a public IP at lookup time and a private IP at the actual TCP connect. `request-filtering-agent` installs as an `https.Agent` that runs the IP check after DNS resolution at connect time, eliminating the TOCTOU window.

Output: Three migrated call sites, deleted safe-fetch.ts + tests, new http-client.ts shared instance, DNS-rebinding test, REVIEW.md marked FIXED.
</objective>

       ```ts
       import axios, { AxiosInstance } from 'axios';
       import { useAgent } from 'request-filtering-agent';

       /**
        * SSRF-protected HTTP client.
        * Replaces the custom safeFetch wrapper from CRIT-07.
        *
        * request-filtering-agent installs an https.Agent that rejects connections
        * to private IP ranges AT CONNECT TIME — this closes the DNS-rebinding /
        * TOCTOU window that an upfront dns.lookup() leaves open.
        *
        * Filter config:
        *   allowPrivateIPAddress: false  → blocks RFC1918 (10/8, 172.16/12, 192.168/16) + CGNAT (100.64/10)
        *   allowLoopbackAddress:  false  → blocks 127/8 + ::1
        *   By default: blocks 0/8, 169.254/16 (link-local incl. AWS IMDS at 169.254.169.254),
        *               224/4 (multicast), 240/4 (reserved), 255.255.255.255, and IPv6 equivalents.
        */
       function makeHttpClient(): AxiosInstance {
         const instance = axios.create({
           timeout: 30_000,
           maxRedirects: 5,
           validateStatus: () => true, // Don't throw on 4xx/5xx — caller decides via response.status.
           // The agent injection happens per-request via useAgent(url) — see helper below.
         });

         instance.interceptors.request.use((config) => {
           const url = config.url;
           if (!url) return config;
           // Inject filtering agent based on URL scheme.
           config.httpAgent = useAgent(url, {
             allowPrivateIPAddress: false,
             allowLoopbackAddress: false,
           });
           config.httpsAgent = useAgent(url, {
             allowPrivateIPAddress: false,
             allowLoopbackAddress: false,
           });
           return config;
         });

         return instance;
       }

       export const httpClient: AxiosInstance = makeHttpClient();

       /**
        * Sentinel error for "URL pointed at a private address and was refused".
        * The library throws Node-level connection errors (ECONNREFUSED-like) — we
        * normalize to a single type so route handlers can map to a generic 400.
        */
       export class BlockedRequestError extends Error {
         constructor(message: string) {
           super(message);
           this.name = 'BlockedRequestError';
         }
       }

       /**
        * Determine whether an axios error was caused by request-filtering-agent rejecting the target.
        * Library throws with code 'ERR_INVALID_URL' or message containing 'is private'.
        */
       export function isBlockedRequestError(err: unknown): boolean {
         if (!err || typeof err !== 'object') return false;
         const e = err as { code?: string; message?: string };
         return (
           e.code === 'ERR_INVALID_URL' ||
           (typeof e.message === 'string' && /private|loopback|link-?local|disallow/i.test(e.message))
         );
       }
       ```

    ```ts
    import { vi } from 'vitest';
    import dns from 'dns';

    describe('httpClient — DNS rebinding fails at connect time (HIGH-13)', () => {
      it('rejects a hostname whose resolution flips to a private IP between checks', async () => {
        // Mock dns.lookup so that the first call returns 1.1.1.1 (public)
        // and subsequent calls return 127.0.0.1 (private).
        let callCount = 0;
        const originalLookup = dns.lookup;
        const lookupSpy = vi.spyOn(dns, 'lookup').mockImplementation((host: any, opts: any, cb: any) => {
          callCount++;
          const callback = typeof opts === 'function' ? opts : cb;
          const ip = callCount === 1 ? '1.1.1.1' : '127.0.0.1';
          // node's dns.lookup callback shape varies; emulate the all=false default shape:
          process.nextTick(() => callback(null, ip, 4));
        });

        try {
          await httpClient.get('http://rebind.example.com/', { timeout: 2000 });
          throw new Error('should have rejected on rebound private IP');
        } catch (err) {
          // request-filtering-agent should reject either the first or the second lookup result.
          // In the worst case (it accepted 1.1.1.1 on first lookup), the connection attempt to
    ```
