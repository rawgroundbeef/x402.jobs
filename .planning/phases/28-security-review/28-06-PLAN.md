<objective>
Remediate HIGH-09 by HMAC-signing the `statusUrl` returned in 202 responses so that anyone who scrapes/leaks the URL (`Referer`, share links, log dumps) cannot read the paid output without the signature. Implement a two-step rollout: API accepts both signed and unsigned URLs for one deploy cycle, then signature becomes required once frontend polling threads it through.

Purpose: The status endpoints today rely on UUID entropy alone. UUIDs leak via referrer headers and share buttons. HMAC adds a viewing token whose secret is not in any client-side surface.

Output: Signing helper, signed 202 responses, dual-acceptance verification on GET, frontend polling updated, REVIEW.md marked FIXED. **autonomous: false** — frontend coordination requires a manual sanity check before flipping the env flag.
</objective>

    ```ts
    import crypto from 'crypto';

    function getSecret(): string {
      const s = process.env.WEBHOOK_SIGNING_SECRET;
      if (!s || s.length < 32) {
        throw new Error('WEBHOOK_SIGNING_SECRET must be set to a 32+ char string');
      }
      return s;
    }

    /** Sign a runId for status URL. expiresAt is Unix seconds. Returns hex string. */
    export function signStatusUrl(runId: string, expiresAt: number): string {
      const payload = `${runId}.${expiresAt}`;
      return crypto.createHmac('sha256', getSecret()).update(payload).digest('hex');
    }

    /** Verify the sig matches runId+exp and exp is in the future. Constant-time compare. */
    export function verifyStatusSignature(
      runId: string,
      sig: string | undefined,
      exp: string | number | undefined,
    ): boolean {
      if (!sig || exp == null) return false;
      const expNum = typeof exp === 'string' ? parseInt(exp, 10) : exp;
      if (!Number.isFinite(expNum) || expNum <= Math.floor(Date.now() / 1000)) {
        return false;
      }
      let expected: string;
      try {
        expected = signStatusUrl(runId, expNum);
      } catch {
        return false; // secret not configured
      }
      const a = Buffer.from(sig, 'utf8');
      const b = Buffer.from(expected, 'utf8');
      if (a.length !== b.length) return false;
      return crypto.timingSafeEqual(a, b);
    }

    /** Build a fully-signed status URL given base URL and runId. */
    export function buildSignedStatusUrl(baseUrl: string, runId: string, ttlSeconds = 86400): string {
      const exp = Math.floor(Date.now() / 1000) + ttlSeconds;
      const sig = signStatusUrl(runId, exp);
      const trimmed = baseUrl.replace(/\/+$/, '');
      return `${trimmed}/status/${runId}?sig=${sig}&exp=${exp}`;
    }
    ```
